CREATE OR REPLACE TRIGGER TRG_MESSAGE_USER_NICK
 AFTER UPDATE OF USER_NICK ON USERTABLE FOR EACH ROW
 BEGIN
     UPDATE MESSAGE 
     SET USER_NICK=:NEW.USER_NICK 
     WHERE USER_NICK=:OLD.USER_NICK;
 END;
